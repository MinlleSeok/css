spacing

  1. letter-spacing: ; 자간 지정
  2. word-spacing: ; 단어간 간격 지정
  3. line-height: ; 행간 지정

letter-spacing:npx/nem/n... ; 자간 지정(음수값 가능)

letter-spacing: 5px; 자간 5px 지정

word-spacing: ; 단어간 간격 지정(음수값 가능)

word-spacing : 30px ; 단어간 간격

word-spacing : 1.5em ; 단어간 간격
word-spacing : -0.3em ; 단어간 간격

line-height: ; 행간 지정

line-height:30px ; 행간 지정
line-height: ; 행간 지정
line-height: ; 행간 지정
